纯前端导出excel(支持多sheet页及表格样式表格合并)

需求

一个奇怪的需求，使用纯前端进行excel导出操作，且支持多sheet页，每个sheet页分别对应一个表格，而且每个表格都需要支持表格行合并（分多级进行行合并，在内容相同则进行合并的基础上低级的列需要在高级列合并的行范围内合并）。

解决过程

一般情况下进行excel导出操作都是后端的事，这次我在网上搜索了很多方法，发现一种可以达到要求。那就是使用sheetjs进行excel导出，该js库是某公司开发完成的。但是这里又有个比较坑的点，那就是这个js库的功能并不是全都开放出来了，如果说想要修改表格的样式，如某个单元格的字体、背景色、是否居中以及行合并、列合并等样式则需要收费使用。这里我又去网上找了一遍，最后发现了一个大佬直接使用sheetjs的社区版本进行导出操作，于是我也沿用了他的方法。sheetjs样式及导出方法

表格合并规则生成

在进行表格导出excel之前，需要进行合并规则生成，而且这个表格的级数不固定，因此需要先编写一个所有表格合并规则的通解生成函数。这里的合并规则是每一级的指标名称相同项进行合并，后续同一级的权重和分数合并规则与指标名称相同，低级别指标名称要在上一级指标名称的合并范围内进行合并。

该合并规则生成的思路是，先以列为单位遍历一列的所有行，如果该行与下一行内容相同则行合并数+1,直到遇到一行与前一行数据不同，则将合并的行数保存到第一个合并格数据中，后续被合并格填充rowspan:0,colspan:0代表该格数据被合并。然后非需要根据规则进行合并的列合并规则保持与前一列相同。

后续低等级列的该合并单元格合并行数加已经合并的行数需要与前一等级列的当前行已合并行数进行比较。如下图所示。

合并行比较示意图

//zbColList指标所在列名组成的数组，arr为数组
function createRowData(arr, zbColList) {
    var mergeRules = []; //合并规则
    var colRules = []; //一列的合并规则
    var colnum = 0; //列数
    for (var num in arr[0]) {
    colnum++;
    }
    //规则初始化
    for (var i = 0; i < arr.length; i++) {
    var tmp = {};
    for (var j in arr[0]) {
        tmp[j] = {
        rowspan: 1,
        colspan: 1
        }
    }
    mergeRules.push(tmp);
    }
    var tmp = [];
    var rowNumTmp = 1; //保存的合并行数
    if (arr) {
    for (var j in arr[0]) {
        for (var i = 0; i < arr.length; i++) {
        if (zbColList.indexOf(j) > 0) {
            if (zbColList.indexOf(j) == zbColList.length - 1) {
            break;
            }
            var rest = 0;
            var last = 0;
            colRules.forEach(function(item, index) {
            rest += item.rowspan;
            });
            for (var q = 0; q <= i; q++) {
            last += mergeRules[q][zbColList[zbColList.indexOf(j) - 1]].rowspan;
            }
            if (arr[i + 1] && arr[i][j] == arr[i + 1][j]) {
            rowNumTmp++;
            if (rowNumTmp + rest > last) {
                colRules.push({
                rowspan: rowNumTmp - 1,
                colspan: 1
                });
                for (var k = 1; k < rowNumTmp - 1; k++) {
                colRules.push({
                    rowspan: 0,
                    colspan: 0
                });
                }
                rowNumTmp = 1;
            }
            } else if (i == arr.length - 1) {
            colRules.push({
                rowspan: rowNumTmp,
                colspan: 1
            });
            for (var k = 1; k < rowNumTmp; k++) {
                colRules.push({
                rowspan: 0,
                colspan: 0
                });
            }
            rowNumTmp = 1;
            colRules.forEach(function(col, colIndex) {
                mergeRules[colIndex][j] = col;
            });
            colRules = [];
            } else {
            colRules.push({
                rowspan: rowNumTmp,
                colspan: 1
            });
            if (rowNumTmp == 2) {
                console.log('inss');
            }

            for (var k = 1; k < rowNumTmp; k++) {
                colRules.push({
                rowspan: 0,
                colspan: 0
                });
            }
            rowNumTmp = 1;
            }
        } else if (zbColList.indexOf(j) == 0) {
            if (arr[i + 1] && arr[i][j] == arr[i + 1][j]) {
            rowNumTmp++;
            } else if (i == arr.length - 1) {
            colRules.push({
                rowspan: rowNumTmp,
                colspan: 1
            });
            for (var k = 1; k < rowNumTmp; k++) {
                colRules.push({
                rowspan: 0,
                colspan: 0
                });
            }
            rowNumTmp = 1;
            colRules.forEach(function(col, colIndex) {
                mergeRules[colIndex][j] = col;
            });
            colRules = [];
            } else {
            colRules.push({
                rowspan: rowNumTmp,
                colspan: 1
            });
            for (var k = 1; k < rowNumTmp; k++) {
                colRules.push({
                rowspan: 0,
                colspan: 0
                });
            }
            rowNumTmp = 1;
            }
        } else {
            var tmpKeys = Object.keys(mergeRules[i]);
            var tmpKey = tmpKeys[tmpKeys.indexOf(j) - 1];
            mergeRules[i][j] = mergeRules[i][tmpKey];
        }
        }
    }
    }
    // console.log('mergeRules!', mergeRules);
    return mergeRules;
}

//获取字符串长度，方便后续导出时设置每列的列宽
function getStrLength(str) {
    var len = str.length;
    var byteLen = 0;
    if (str) {
    for (var i = 0; i < len; i++) {
        if (str.charCodeAt(i) > 255) {
            byteLen += 2;//ascii码大于255则为两位字符，如中文
        } else {
            byteLen++;//小于则为一位字符
        }
    }
    }
    return byteLen;
}

function getCharCol(n) {
    let temCol = '',
    s = '',
    m = 0
    while (n > 0) {
    m = n % 26 + 1
    s = String.fromCharCode(m + 64) + s
    n = (n - m) / 26
    }
    return s
}

function s2ab(s) {
    if (typeof ArrayBuffer !== 'undefined') {
    var buf = new ArrayBuffer(s.length);
    var view = new Uint8Array(buf);
    for (var i = 0; i != s.length; ++i) view[i] = s.charCodeAt(i) & 0xFF;
    return buf;
    } else {
    var buf = new Array(s.length);
    for (var i = 0; i != s.length; ++i) buf[i] = s.charCodeAt(i) & 0xFF;
    return buf;
    }
}

//arr表格数据,rulesList合并规则列表,目前该方法仅支持表格列数小于等于26时的导出
function downloadExl(arr, rulesList) {
    //深拷贝
    var jsonsList = JSON.parse(JSON.stringify(arr));
    var resultList = [];
    var outputPosList = [];
    jsonsList.forEach((item, index) => {
        var rules = rulesList[index];
        var jsons = [...item.point];
        var tmpdata = { ...jsons[0] };
        var keyMap = [];//获取keys
        var colWidth = [];//列宽数组
        jsons.unshift({});
        for (var k in tmpdata) {
            var q = '';
            keyMap.push(k);
            colWidth.push(0);//列宽数组初始化
            switch (k) {
                case 'YJZB':
                    q = '一级指标';
                    break;
                case 'YJQZ':
                    q = '一级权重';
                    break;
                case 'YJFS':
                    q = '一级分数';
                    break;
                case 'EJZB':
                    q = '二级指标';
                    break;
                case 'EJQZ':
                    q = '二级权重';
                    break;
                case 'EJFS':
                    q = '二级分数';
                    break;
                case 'SJZB':
                    q = '三级指标';
                    break;
                case 'SJQZ':
                    q = '三级权重';
                    break;
                case 'SJFS':
                    q = '三级分数';
                    break;
                case 'SIJZB':
                    q = '四级指标';
                    break;
                case 'SIJQZ':
                    q = '四级权重';
                    break;
                case 'SIJFS':
                    q = '四级分数';
                    break;
                case 'TSNKCS':
                    q = '特殊内控措施';
                    break;
            }
            jsons[0][k] = q;
        }
        jsons.unshift({});
        for (var k in tmpdata) {
            switch (k) {
                case 'YJZB':
                    jsons[0][k] = '任务名称';
                    break;
                case 'YJQZ':
                    jsons[0][k] = item.task_name;
                    break;
                case 'YJFS':
                    jsons[0][k] = '评估主体';
                    break;
                case 'EJZB':
                    jsons[0][k] = item.primary;
                    break;
                case 'EJQZ':
                    jsons[0][k] = '评估单位';
                    break;
                case 'EJFS':
                    jsons[0][k] = item.unit;
                    break;
                default:
                    jsons[0][k] = '';

            }
        }
        for (var k in jsons[0]) {
            var tmpWidth = getStrLength(jsons[0][k]);
            for (var i = 1; i < jsons.length; i++) {
                var currentWidth = getStrLength(jsons[i][k]);
                if (currentWidth > tmpWidth) {
                    tmpWidth = currentWidth;
                }
                if (i == jsons.length - 1) {
                    colWidth[keyMap.indexOf(k)] = tmpWidth;
                }

            }
        }

        var tmpdata = [];
        jsons.map((v, i) => keyMap.map((k, j) => Object.assign({}, {
            v: v[k],
            position: (j > 25 ? getCharCol(j) : String.fromCharCode(65 + j)) + (i + 1)
        }))).reduce((prev, next) => prev.concat(next)).forEach((v, i) => tmpdata[v.position] = {
            v: v.v
        });
        var outputPos = Object.keys(tmpdata); //设置区域,比如表格A1到D10
        outputPosList.push(outputPos);
        tmpdata["!merges"] = [];//设置行合并
        rules.forEach((item, index) => {
            for (var k in item) {
                if (item[k].rowspan != 0 && item[k].rowspan != 1) {
                    tmpdata["!merges"].push({
                        s: {
                            c: keyMap.indexOf(k),
                            r: index + 2
                        },
                        e: {
                            c: keyMap.indexOf(k),
                            r: index + item[k].rowspan + 1
                        }
                    });
                }
            }

        });
        for (var key in tmpdata) {
            if (key.substr(0, 1) != "!" && key.substr(1)) {
                tmpdata[key].s = {
                    border: {
                        top: {
                            style: 'thin',
                            color: {
                                rgb: "000000"
                            }
                        },
                        left: {
                            style: 'thin',
                            color: {
                                rgb: "000000"
                            }
                        },
                        bottom: {
                            style: 'thin',
                            color: {
                                rgb: "000000"
                            }
                        },
                        right: {
                            style: 'thin',
                            color: {
                                rgb: "000000"
                            }
                        }
                    },
                    alignment: {
                        vertical: "center",
                        wrapText: true
                    }
                };
            }
        }
        tmpdata["!cols"] = [];//表格列宽配置
        colWidth.forEach((item, index) => {
            tmpdata["!cols"].push({
                wch: item
            });

        });
        resultList.push(tmpdata);
    });
    var tmpWB = {
        SheetNames: [],
        Sheets: {

        }
    };
    resultList.forEach((item, index) => {
        var sheetName = 'Sheet' + (index + 1);//这里可以自定义sheet页的名称，但是要保证sheet页名称唯一，否则会出错
        tmpWB.SheetNames.push(sheetName);
        tmpWB.Sheets[sheetName] = Object.assign({}, item, {
            '!ref': outputPosList[index][0] + ':' + outputPosList[index][outputPosList[index].length - 1]//设置区域
        });
    });
    tmpDown = new Blob([s2ab(XLSX.write(tmpWB, {
        bookType: 'xlsx',
        bookSST: false,
        type: 'binary'
    } //这里的数据是用来定义导出的格式类型
    ))], {
        type: ""
    });
    var wopts = {
        bookType: 'xlsx',
        bookSST: true,
        type: 'binary',
        cellStyles: true
    };
    //这里可以自定义导出的文件名
    saveAs(tmpDown, "评估结果." + (wopts.bookType == "biff2" ? "xls" : wopts.bookType));
}
